\introduction{%
Eine klassische, süß-salzige Teriyaki-Sauce mit leichter Schärfe von Ingwer und Knoblauch. Perfekt zum Glasieren von Gemüse, Tofu oder als Dip.
}

\ingredients{
    100 g & Brauner Zucker\\
    250 ml & Sojasauce\index{Würzmittel!Sojasauce}\\
    30 ml & Reisessig\\
    5 g & Ingwer\index{Gemüse!Ingwer}\\
    4 Zehen & Knoblauch\index{Gemüse!Knoblauch}\\
    10 ml & Sesamöl\\
    24 g & Maisstärke\index{Getreide!Maisstärke}\\
    60 ml & Wasser\\
    9 g & Sesamsamen\index{Nüsse \& Samen!Sesam}
}

\preparation{%
    \step%
    Zucker und Sojasauce bei niedriger Hitze unter Rühren erhitzen, bis sich der Zucker vollständig aufgelöst hat.
    
    \step%
    Reisessig, Ingwer und Knoblauch zugeben und kurz köcheln lassen.
    
    \step%
    Sesamöl einrühren.\\
    
    \step%
    Maisstärke mit Wasser glatt rühren und unter ständigem Rühren langsam einarbeiten, bis die Sauce eindickt. Vom Herd ziehen.
    
    \step%
    Sesamsamen in einer Pfanne ohne Öl rösten und unterheben oder darüberstreuen.
}
