\begin{recipe}
[
    preparationtime = {\unit[50]{min}},
    portion = {\portion{2}},
    calory = {1040 kcal | 28g Protein | 115g KH | 48g Fett},
    rating = {4}
]
{Karamellisierte Zwiebel Pasta}

    \introduction{%
        Süßlich, würzig und cremig durch Kokosmilch und Hefeflocken. 
        Perfekt wenn man Lust auf intensive Zwiebel-Aromen hat.
    }

    \ingredients{%
        1 EL & Olivenöl\\
        1 EL & Margarine\\
        2 & Große Zwiebeln\\
        30 g & Getrocknete Tomaten\\
        1 TL & Chili in Öl\\
        \nicefrac{1}{2} TL & Oregano\\
        \nicefrac{1}{2} TL & Basilikum\\
        \nicefrac{1}{2} TL & Petersilie\\
        \nicefrac{1}{4} TL & Thymian\\
        \nicefrac{1}{4} TL & Rosmarin\\
        \nicefrac{1}{4} TL & Majoran\\
        1 EL & Sojasauce\\
        \nicefrac{1}{2} EL & Balsamico\\
        150 ml & Kokosmilch\\
        25 g & Hefeflocken\\
        4 & Knoblauchzehen\\
        200 g & Nudeln
    }

    \preparation{%
        \begin{enumerate}
            \item Zwiebeln in feine Halbringe schneiden und mit Öl und Margarine ca. 35 Minuten karamellisieren, regelmäßig rühren, am Schluss mehr rühren.
            \item Knoblauch und getrocknete Tomaten fein schneiden.
            \item 30 Minuten nach Kochbeginn Nudeln in Salzwasser kochen.
            \item Restliche Zutaten (außer Kokosmilch und Hefeflocken) zu den Zwiebeln geben und kurz (4 Minuten) anbraten.
            \item Kokosmilch und etwas Nudelwasser zugeben und einige Minuten köcheln lassen.
            \item Hefeflocken unterrühren.
            \item Nudeln unterheben und servieren.
        \end{enumerate}
    }

    \hint{
        Wenn es cremiger sein soll, etwas länger köcheln lassen oder leicht pürieren.
    }

\end{recipe}
