\begin{recipe}[
    preparationtime = {1h 30min},
    bakingtime = {50min},
    bakingtemperature = {\protect\bakingtemperature{
        topbottomheat=\unit[200]{\textcelcius}
    }},
    portion = {\portion{8}},
    calory = {780kcal | 14g Protein | 52g KH | 55g Fett},
    price = {2,62€},
    rating = {4},
    source = {\protect\recipesource{https://cooking.nytimes.com/recipes/1020596-vegetarian-mushroom-wellington}{NYTimes Cooking: Vegetarian Mushroom Wellington}}
]{Mushroom Wellington}

\graph{% pictures
        small=pic/hauptgericht/07_veggi-beef-wellington-small,
        big=pic/hauptgericht/07_veggi-beef-wellington-big
    }

\ingredients{
    \textbf{Pilze} & \\
    4 & Portobello Pilze\\
    900g & Gemischte Pilze\\
    200ml & Olivenöl (gesamt)\\
    4 & Schalotten\\
    6 & Knoblauchzehen\\
    2 EL & Rosmarin\\
    1 TL & Thymian\\
    80ml & Portwein\\
    2 EL & Sojasauce\\
    100g & Walnüsse\\
    \\
    \textbf{Zwiebeln} & \\
    3 EL & Butter\\
    2 & Gelbe Zwiebeln\\
    240ml & Apfelsaft\\
    1 EL & Balsamico (optional)\\
    \\
    \textbf{Teig} & \\
    400g & Blätterteig\\
    1 & Ei\\
    Mehl & zum Bestäuben\\
    \multicolumn{2}{l}{\textit{\small Fortsetzung auf nächster Seite}}
}
\ingredientsB{
    \textbf{Sauce} & \\
    360ml & Portwein\\
    360ml & Gemüsebrühe\\
    3 EL & Butter\\
    1 & Schalotte\\
    2 & Knoblauchzehen
}

\preparation{
    \begin{enumerate}
        \item Portobellos putzen, mit Öl bepinseln und je 5min pro Seite anbraten. Abkühlen lassen.
        \item Gemischte Pilze fein hacken.
        \item In Olivenöl Pilze, Schalotten, Knoblauch und Kräuter 10min kräftig anbraten.
        \item Port und Sojasauce zugeben und einkochen lassen. Walnüsse unterrühren und abkühlen.
        \item Zwiebeln mit Butter, Zucker und Gewürzen karamellisieren, Apfelsaft zugeben und einkochen.
        \item Ofen auf 200°C vorheizen. Blätterteig ausrollen.
        \item Hälfte der Pilzmasse mittig verteilen, Zwiebeln darauf, Portobellos darauflegen.
        \item Restliche Pilzmasse darüber geben und zu einer kompakten Form drücken.
        \item Blätterteig darüber schlagen, mit Ei bestreichen und verschließen.
        \item 45–50min backen bis goldbraun.
        \item Für die Reduktion Port und Brühe einkochen, Butter einrühren.
    \end{enumerate}
}

\suggestion[Festlich]{Perfekt als vegetarisches Weihnachts-Hauptgericht mit Rotkohl und Kartoffelklößen.}

\hint{Die Pilzmasse muss vollständig abgekühlt sein, bevor sie in den Blätterteig kommt – sonst wird der Teig weich.}

\end{recipe}
