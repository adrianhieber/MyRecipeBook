\begin{recipe}
[
    preparationtime = {\unit[15]{min}},
    bakingtime = {\unit[40--90]{min}},
    portion = {\portion{2}},
    calory = {680kcal (1030kcal) | 33g Protein (46g) | 41g KH (116g) | 43g Fett (44g)},
    rating = {3},
    price = {3,13€ (3,51€)}
]
{Bolognese}

\introduction{%
    Aromatische, leicht süßliche Gemüse-Bolognese mit dunkler Schokolade 
    für mehr Tiefe. Funktioniert perfekt mit Pasta.
}

\ingredients{%
    1 Block & Räuchertofu \\
    1 & Zwiebel\\
    100 g & Möhre\\
    100 g & Sellerie\\
    200 g & Datteltomaten\\
    200 g & Passierte Tomaten\\
    85 ml & Rotwein\\
    150 ml & Gemüsebrühe\\
    1 EL & Tomatenmark\\
    1 Stück & Dunkle Schokolade (70\%)\\
    1 TL & Zucker\\
    1 TL & Petersilie\\
    1 TL & Tofugewürz\\
    1 TL & Ital. Kräuter\\
    \nicefrac{1}{2} TL & Salz\\
    2 EL & Olivenöl
}

\preparation{%
    \step%
    Zwiebel, Möhre und Sellerie fein schneiden und im Olivenöl anbraten.
    
    \step%
    Tofu fein raspeln, in Topf zugeben und circa 4 Minuten mit anbraten
    
    \step%
    Tomatenmark kurz mitrösten.\\
    
    \step%
    Halbierte Datteltomaten zugeben und 4 weitere Minuten kochen.
    
    \step%
    Mit Brühe ablöschen, restliche Zutaten zufügen.\\
    
    \step%
    40 Minuten (oder länger) köcheln lassen.\\
    
    \step%
    Beilage (Pasta) anfangen zu kochen.\\
    
    \step%
    Abschmecken und mit Pasta servieren.
}

\suggestion[Mit Pasta]{%
    Am besten mit 100g Spaghetti pro Person servieren.
}

\hint{%
    Je länger die Sauce köchelt, desto intensiver wird der Geschmack. Die Schokolade mildert die Säure der Tomaten.
}

\end{recipe}
