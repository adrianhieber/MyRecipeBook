\begin{recipe}[
    preparationtime = {10 min},
    portion = {\portion{2}},
    calory = {430kcal | 6g Protein | 34g Fett | 26g Kohlenhydrate},
    price = {2,10€},
    rating = {1},
    source = {Adrian Hieber}
]{Acili Ezme}

\ingredients{
    2 & Tomaten\\
    1 & Zwiebel\\
    2 & Rote Spitzpaprika\\
    2 & Grüne Spitzpaprika\\
    1--2 & Knoblauchzehen\\
    1 EL & Paprikamark\\
    1 EL & Tomatenmark\\
    1 EL & Sumak\\
    \nicefrac{1}{2} & Zitrone (Saft)\\
    3 EL & Olivenöl\\
    3 EL & Granatapfelsirup\\
    & Salz, Pfeffer
}

\preparation{%
    \begin{enumerate}
        \item Alles grob zerkleinern und zusammen pürieren, bis eine dicke, aber nicht komplett flüssige Masse entsteht.\\
        \item Die Masse durch ein Sieb oder mit einem Tuch gut ausdrücken, damit überschüssiger Saft entfernt wird.\\
        \item Mindestens 1 Stunde im Kühlschrank ziehen lassen.
    \end{enumerate}
}

\suggestion[Serviervorschlag]{Mit Fladenbrot, als Grillbeilage oder zu Mezze servieren. Mit etwas zusätzlichem Olivenöl und Petersilie toppen.}

\end{recipe}
