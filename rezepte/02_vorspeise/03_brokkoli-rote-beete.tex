\begin{recipe}[
    preparationtime = {20 min},
    portion = {\portion{2}},
    calory = {720kcal | 22g Protein | 48g KH | 48g Fett},
    price = {2,89€},
    rating = {1}
]{Brokkoli auf Rote-Beete-Hummus}

\graph{% pictures
        big=pic/vorspeise/03_brokkoli-rote-beete  % big picture
    }

\ingredients{
    1 & Großer Brokkoli\\
    2--3 & Rote Beete (vorgekocht)\\
    500g & Kichererbsen\\
    4 EL & Tahin\\
    1 & Zitrone\\
    2 & Knoblauchzehen\\
    2 EL & Balsamico\\
    3 EL & Olivenöl\\
    1 TL & Kreuzkümmel\\
    2 EL & Petersilie\\
    & Salz, Pfeffer
}

\preparation{
    \begin{enumerate}
        \item Brokkoli in 1cm dicke Scheiben schneiden und in etwas Olivenöl goldbraun anbraten.
        \item Für den Hummus Rote Beete, Kichererbsen (mit 2 EL Saft), Tahin, Kreuzkümmel, Knoblauch und Zitronensaft pürieren.
        \item Eiswürfel zugeben und kurz weiter pürieren, bis die Konsistenz cremig ist.
        \item Olivenöl mit Balsamico vermischen.
        \item Petersilie fein hacken.
        \item Hummus auf einem Teller verstreichen, Brokkoli darauf legen und mit Topping und Petersilie garnieren.
    \end{enumerate}
}

\suggestion[Schärfe]{Etwas Chili oder Harissa ins Balsamico-Öl geben.}

\hint{Ein Eiswürfel im Hummus macht ihn besonders luftig und cremig.}

\end{recipe}
