\begin{recipe}
[
    preparationtime = {\unit[20]{min}},
    portion = {\portion{2}},
    calory = {260 kcal | 5g Protein | 32g KH | 11g Fett},
    rating = {4},
    source = {Nico Skerra}
]
{Zwiebelsuppe}

\introduction{%
    Auch eines der Rezepte, welches ich von Freunden bekommen habe.
}

\ingredients{%
    2 & Große Zwiebeln\\
    1 EL & Mehl\\
    50 ml & Weißwein\\
    1 EL & Tomatenmark\\
    400 ml & Gemüsebrühe\\
    10 g & Butter oder Olivenöl\\
    & Salz\\
    & Pfeffer\\
    (optional) & Petersilie\\
    (optional) & 1 EL Zucker
}

\preparation{%
    \begin{enumerate}
        \item Butter oder Öl erhitzen.
        \item Zwiebeln in Halbringe schneiden und langsam anschwitzen, bis sie leicht bräunen und Bratensatz entsteht, hier eventuell den Zucker zufügen.
        \item Tomatenmark unterrühren, dann Mehl einarbeiten.
        \item Mit Weißwein ablöschen und etwas einkochen lassen.
        \item Mit Gemüsebrühe aufgießen und einige Minuten köcheln.
        \item Mit Salz, Pfeffer und optional Petersilie abschmecken.
    \end{enumerate}
}

\suggestion[Upgrade]{%
    Mit geröstetem Brot und etwas Käse wird daraus eine schnelle Gratinsuppe.
}

\hint{%
    Zwiebeln nicht zu schnell bräunen, mittlere Hitze gibt mehr Süße und weniger Bitterkeit.
}

\end{recipe}
