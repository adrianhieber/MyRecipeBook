\begin{recipe}
[% 
    preparationtime = {\unit[45]{min}},
    portion = {\unit[150]{g} Sirup (ca. 6 Drinks)},
    calory = {50 kcal | 0g Protein | 12g KH | 0g Fett},
    rating = {5},
    price = {0,26€},
    source = {\protect\recipesource{https://www.youtube.com/watch?v=9uBnJAEiDug\&t=458s}{YouTube: My Name is Andong}}
]
{Club Mate Sirup}
    
    \graph{% pictures
        big=pic/getraenke/01_club-mate
    }
    
    \introduction{%
        Gar nicht vergleichbar mit dem Orginal, schmeckt viel natürlicher und beerig, mehr wie ein süßes Sportgetränk und nicht wie ausgedrückte Zigaretten. Aber aufpassen, der Koffeingehalt ist nicht zu unterschätzen.
    }
    
    \ingredients{%
        100 g & Mateblätter (grob)\\
        500 g & Wasser\\
        75 g & Zucker\\
        3{,}5 g & Zitronensäure
    }
    
    \preparation{%
        \begin{enumerate}
            \item Wasser aufkochen, vom Herd nehmen und die Mateblätter hinzufügen. 
                  Ohne weitere Hitze etwa 10 Minuten ziehen lassen.
            
            \item Zuerst durch ein grobes Sieb filtern, anschließend durch Küchenpapier 
                  oder optional einen Kaffeefilter fein filtern.
            
            \item Zucker hinzufügen und erneut aufkochen. 
                  Dann bei mittlerer Hitze reduzieren lassen, 
                  bis etwa \unit[150]{g} Sirup übrig sind 
                  (ca. 20 Minuten).
            
            \item Zitronensäure einrühren und vollständig abkühlen lassen.
            
            \item In eine saubere Flasche abfüllen und kühl lagern.
        \end{enumerate}
    }
    
    \suggestion{%
        Mischverhältnis 1:10 mit Sprudelwasser 
        (z.B. 25 g Sirup mit 225 ml Wasser).
    }

    \suggestion[Koffeinmenge]{%
        \begin{itemize}
            \item 1gr Sirup enthalten ca 8mg Koffein
            \item 250ml Getränk enthalten ca 180mg Koffein
            \item 1 Espresso enthält 60--80mg Koffein
        \end{itemize}
    }

    
\end{recipe}
