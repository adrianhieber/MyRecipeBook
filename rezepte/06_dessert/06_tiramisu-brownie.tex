\begin{recipe}[
    preparationtime = {1h},
    bakingtime = {25min},
    bakingtemperature = {\protect\bakingtemperature{
        topbottomheat=\unit[180]{\textcelcius}
    }},
    portion = {\portion{8}},
    calory = {900kcal | 13g Protein | 69g KH | 65g Fett},
    price = {1,96€},
    rating = {1},
    source = {\protect\recipesource{https://kitchen-by-the-sea.com/tiramisu-brownies/\#recipe}{Kitchen by the sea: Tiramisu Brownie}}
]{Tiramisu Brownies}

\ingredients{
    \textbf{Brownies} & \\
    227g & Butter\\
    400g & 70\% Zartbitterschokolade\\
    60g & Kakaopulver\\
    120g & Mehl\\
    5 & Eier\\
    140g & Brauner Zucker\\
    200g & Zucker\\
    1 TL & Salz\\
    1 EL & Vanilleextrakt\\
    1 EL & Instant Espresso-Pulver\\
    \\
    \textbf{Creme} & \\
    250g & Mascarpone\\
    240ml & Sahne\\
    50g & Zucker\\
    \nicefrac{1}{4} TL & Salz\\
    1 EL & Vanille\\
    1 EL & Instant Espresso-Pulver\\
    \\
    \textbf{Topping} & \\
    3 EL & Kakaopulver
}

\preparation{
    \begin{enumerate}
        \item Backofen auf 175°C vorheizen. Form einfetten und mit Backpapier auslegen.
        \item Butter bei niedriger Hitze schmelzen, Schokolade einrühren und vollständig schmelzen lassen. Abkühlen lassen.
        \item Mehl, Kakao und Espresso sieben.
        \item Eier, Zucker, Salz und Vanille 4–5min cremig aufschlagen.
        \item Geschmolzene Schokolade unterheben.
        \item Mehlmischung vorsichtig unterheben.
        \item Teig in die Form geben und 25min backen. Komplett auskühlen lassen.
        \item Für die Creme Mascarpone und Sahne kurz verrühren.
        \item Zucker, Salz, Vanille und Espresso zugeben und auf mittlerer Stufe zu soft peaks schlagen.
        \item Creme auf den Brownies verteilen und mit Kakao bestäuben.
    \end{enumerate}
}

\suggestion[Servieren]{Über Nacht im Kühlschrank durchziehen lassen – dadurch wird die Textur noch tiramisu-artiger.}

\suggestion{Ich habe es noch nich gekocht.}

\hint{Nicht zu lange backen – die Mitte darf noch leicht weich wirken, sonst werden sie trocken.}

\end{recipe}
