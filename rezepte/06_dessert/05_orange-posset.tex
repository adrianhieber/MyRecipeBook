\begin{recipe}[
    preparationtime = {15min},
    bakingtime = {\unit[3]{h}},
    portion = {\portion{4}},
    calory = {360kcal | 3g Protein | 25g KH | 28g Fett},
    price = {1,02€},
    rating = {1}
]{Orangen-Possets}

\ingredients{
    400 g & Sahne\\
    6 & Orangen\\
    90 g & Zucker\\
    4 TL & Orangenschale\\
    Prise & Salz\\
    Prise & Kurkuma
}

\preparation{
    \begin{enumerate}
        \item Orangenschale abreiben und Orangen auspressen.
        \item Orangen halbieren (von Strunk zum Boden nicht seitlich)
        \item In einem Topf bei mittlerer Hitze Sahne, Zucker, Kurkuma und Orangenschale erhitzen.
        \item 5min köcheln lassen, dann von der Hitze nehmen.
        \item Orangensaft zugeben und gut verrühren.
        \item In Orangenschalen oder Förmchen füllen und mindestens 3h im Kühlschrank fest werden lassen.
    \end{enumerate}
}

\suggestion[Servieren]{Mit etwas geriebener Schokolade oder Orangenabrieb toppen.}

\hint{Nicht kochen lassen, sobald der Saft zugegeben wurde, sonst kann die Sahne ausflocken. Man kann bisschen Agar Agar zufügen, damit es fester wird.}

\end{recipe}
