\begin{recipe}[
    preparationtime = {15min},
    bakingtime = {\unit[3]{h}},
    portion = {\portion{4}},
    calory = {360kcal | 3g Protein | 25g KH | 28g Fett},
    price = {1,02€},
    source = {\protect\recipesource{https://www.instagram.com/p/C3kulRlujU5/}{Instagram: crowded\_kitchen}},
    rating = {1}
]{Orangen-Possets}

\ingredients{
    400 g & Sahne\\
    6 & Orangen\\
    90 g & Zucker\\
    4 TL & Orangenschale\\
    Prise & Salz\\
    Prise & Kurkuma
}

\preparation{
    \step%
    Orangenschale abreiben und Orangen auspressen.
    
    \step%
    Orangen halbieren (von Strunk zum Boden nicht seitlich)
    
    \step%
    In einem Topf bei mittlerer Hitze Sahne, Zucker, Kurkuma und Orangenschale erhitzen.
    
    \step%
    5min köcheln lassen, dann von der Hitze nehmen.
    
    \step%
    Orangensaft zugeben und gut verrühren.
    
    \step%
    In Orangenschalen oder Förmchen füllen und mindestens 3h im Kühlschrank fest werden lassen.
}

\suggestion[Servieren]{Mit etwas geriebener Schokolade oder Orangenabrieb toppen.}

\hint{Nicht kochen lassen, sobald der Saft zugegeben wurde, sonst kann die Sahne ausflocken. Man kann bisschen Agar Agar zufügen, damit es fester wird.}

\end{recipe}
