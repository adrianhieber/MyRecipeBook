\begin{recipe}
[% 
    preparationtime = {\unit[10]{min}},
    portion = {\portion{2}},
    source = {\url{}},
    source = {\protect\recipesource{https://www.brigitte.de/rezepte/koch-trends/desserts--karamellisierte-feigen--fruchtig-geniessen-10091476.html}{Brigitte.de: Karaellisierte Feigen}},
    calory = {450 kcal | 4g Protein | 65g KH | 17g Fett},
    rating = {5}
]
{Karamellisierte Feigen}
    
    \graph{% pictures
        big=pic/dessert/02_karamelisierte-feigen
    }
    
    \introduction{%
        Warme, karamellisierte Feigen mit einer reduzierten Portwein und kaltem Vanilleeis sind eine mega Kombi. Das ist mein schnelles Go-to Dessert wenn es fancy werden soll.
    }
    
    \ingredients{%
        6 & Feigen\\
        8 g & Butter\\
        4 TL & Brauner Zucker\\
        75 ml & Roter Portwein\\
        1 TL & Balsamico-Essig\\
        2 Kugeln & Vanilleeis\\
        Optional & Schuss Zitronen- oder Orangensaft
    }
    
    \preparation{%
        \begin{enumerate}
            \item Butter mit 2 TL Zucker in einer Pfanne schmelzen lassen.
            
            \item Feigen halbieren und den Strunk entfernen.
            
            \item Mit der Schnittfläche nach unten in die Pfanne legen.
            
            \item Restlichen Zucker auf die Außenseite der Feigen geben.
            
            \item 3--4 Minuten karamellisieren lassen, dann wenden. 
                  Kurz warten (2 min) und nochmals wenden, bis sie schön gebräunt sind.

            \item Mit Portwein ablöschen und etwa 2 Minuten einkochen lassen.
            
            \item Feigen aus der Pfanne nehmen und auf Teller platzieren.
            
            \item Balsamico (optional mit etwas Zitronen- oder Orangensaft) 
                  in den Jus geben und 1--2 Minuten reduzieren lassen.
            
            \item Jus auf Feigen geben 
                  und mit Vanilleeis servieren.
        \end{enumerate}
    }
    
\end{recipe}
