\begin{recipe}
[
    preparationtime = {\unit[10]{min}},
    bakingtime = {\unit[50--55]{min}},
    bakingtemperature = {\protect\bakingtemperature{
        topbottomheat=\unit[180]{\textcelcius}
    }},
    portion = {\portion{12}},
    calory = {220 kcal | 4g Protein | 30g KH | 9g Fett},
    rating = {3},
    source = {Adrian Hieber},
    price = {0,22€}
]
{Bananenbrot}

\introduction{%
    Perfekt um überreife Bananen zu retten. Saftig, schokoladig und schnell zusammengerührt.
}

\ingredients{%
    3 & Reife Bananen\\
    72 ml & Neutrales Öl\\
    110 g & Brauner Zucker\\
    200 g & Mehl\\
    2 & Eier\\
    \nicefrac{1}{2} Tafel & Schokolade (ca. 50 g)\\
    \nicefrac{1}{2} Pck. & Vanillezucker\\
    2 TL & Backpulver\\
    Prise & Zimt\\
    Prise & Salz
}

\preparation{%
    \step%
    Bananen mit einer Gabel fein zerdrücken.
    
    \step%
    Eier, Öl und Zucker unterrühren.
    
    \step%
    Mehl, Backpulver, Salz und Gewürze mischen und unterheben.
    
    \step%
    Schokolade grob hacken und unterziehen.
    
    \step%
    In eine gefettete und gemehlte Kastenform füllen.
    
    \step%
    Bei 180°C ca. 50--55 Minuten backen.
}

\suggestion{
    Auch mit gefrorenen Früchten statt Schokolade möglich.
}

\end{recipe}
